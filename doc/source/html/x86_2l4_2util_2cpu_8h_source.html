71  "pushfl \t\n"
72  "popl %%eax \t\n" /* get eflags */
73  "movl %%eax, %%ebx \t\n" /* save it */
74  "xorl $0x200000, %%eax \t\n" /* toggle ID bit */
75  "pushl %%eax \t\n"

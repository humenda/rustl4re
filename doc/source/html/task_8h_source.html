288 L4_INLINE l4_msgtag_t
289 l4_task_map_u(l4_cap_idx_t dst_task, l4_cap_idx_t src_task,
290  l4_fpage_t snd_fpage, unsigned long snd_base, l4_utcb_t *u) L4_NOTHROW
291 {
292  l4_msg_regs_t *v = l4_utcb_mr_u(u);
293  v->mr[0] = L4_TASK_MAP_OP;
294  v->mr[3] = l4_map_obj_control(0,0);
295  v->mr[4] = l4_obj_fpage(src_task, 0, L4_FPAGE_RWX).raw;
296  v->mr[1] = snd_base;
297  v->mr[2] = snd_fpage.raw;
298  return l4_ipc_call(dst_task, u, l4_msgtag(L4_PROTO_TASK, 3, 1, 0), L4_IPC_NEVER);
299 }

312 L4_INLINE l4_msgtag_t
313 l4_task_unmap_batch_u(l4_cap_idx_t task, l4_fpage_t const *fpages,
314  unsigned num_fpages, unsigned long map_mask,
315  l4_utcb_t *u) L4_NOTHROW
316 {
317  l4_msg_regs_t *v = l4_utcb_mr_u(u);
318  v->mr[0] = L4_TASK_UNMAP_OP;
319  v->mr[1] = map_mask;
320  __builtin_memcpy(&v->mr[2], fpages, num_fpages * sizeof(l4_fpage_t));
321  return l4_ipc_call(task, u, l4_msgtag(L4_PROTO_TASK, 2 + num_fpages, 0, 0), L4_IPC_NEVER);
322 }

537 L4_INLINE int
538 l4_sndfpage_add_u(l4_fpage_t const snd_fpage, unsigned long snd_base,
539  l4_msgtag_t *tag, l4_utcb_t *utcb) L4_NOTHROW
540 {
541  l4_msg_regs_t *v = l4_utcb_mr_u(utcb);
542  int i = l4_msgtag_words(*tag) + 2 * l4_msgtag_items(*tag);
543 
544  if (i >= L4_UTCB_GENERIC_DATA_SIZE - 1)
545  return -L4_ENOMEM;
546 
547  v->mr[i] = snd_base | L4_ITEM_MAP | L4_ITEM_CONT;
548  v->mr[i + 1] = snd_fpage.raw;
549 
550  *tag = l4_msgtag(l4_msgtag_label(*tag), l4_msgtag_words(*tag),
551  l4_msgtag_items(*tag) + 1, l4_msgtag_flags(*tag));
552  return 0;
553 }

587 Path::cmp_path(char const *n) const throw()
588 {
589  char const *p = _p;
590  for (; *p && !__is_sep(*p) && *n; ++p, ++n)
591  if (*p != *n)
592  return Path();
593 
594  if (*n || (*p && !__is_sep(*p)))
595  return Path();
596

624 inline Path
625 Mount_tree::lookup(Path const &path, cxx::Ref_ptr<Mount_tree> *mt,
626  cxx::Ref_ptr<Mount_tree> *mp) throw()
627 {
628  cxx::Ref_ptr<Mount_tree> x(this);
629  Path p = path;
630 
631  if (p.first().cmp_path("."))
632  p.strip_first();
633 
634  Path last_mp = p;
635 
636  if (mp)
637  *mp = x;;
638 
639  while (1)
640  {
641  Path r = x->find(p, &x);
642 
643  if (!r)
644  {
645  if (mp)
646  return last_mp;
647 
648  if (mt)
649  *mt = x;
650 
651  return p;
652  }
653 
654  r.strip_sep();
655 
656  if (mp && x->_mount)
657  {
658  last_mp = r;
659  *mp = x;
660  }
661 
662  if (r.empty())
663  {
664  if (mt)
665  *mt = x;
666 
667  if (mp)
668  return last_mp;
669  else
670  return r;
671  }
672 
673  p = r;
674  }
